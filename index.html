#!/usr/bin/env sh _exists() { cmd="$1" if [ -z "$cmd" ] ; then echo "Usage: _exists cmd" return 1 fi if type command >/dev/null 2>&1 ; then command -v $cmd >/dev/null 2>&1 else type $cmd >/dev/null 2>&1 fi ret="$?" return $ret } if _exists curl && [ "${ACME_USE_WGET:-0}" = "0" ]; then curl https://raw.githubusercontent.com/Guakamoli/get.tunnel.paiya.app/main/tunnel.sh | sh -s -- elif _exists wget ; then wget -O - https://raw.githubusercontent.com/Guakamoli/get.tunnel.paiya.app/main/tunnel.sh | sh -s -- else echo "Sorry, you must have curl or wget installed first." echo "Please install either of them and try again." fi
